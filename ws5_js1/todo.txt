Jelenleg nem szükséges mélyebben tanulmányoznia.

HTML

A 7. feladatrésznél egy táblázatot kell feltöltenie, ehhez segítség a következő példa:
let values = {
    "Frontend Developer": 200000,
    "Backend Developer": 600000,
    "Full Stack Developer": 800000
}
function loadTableData(values) {  
    const table = document.getElementById('tableId');
    table.innerHTML = "";
    for (let value in values) {
        const row = table.insertRow();
        const cell0 = row.insertCell(0);
        const cell1 = row.insertCell(1);
        cell0.textContent = value;
        cell1.textContent = values[value];
        cell1.classList.add('customCell'); // add a class for later styling
    }
}
loadTableData(values);

A táblázat celláinak formázása is lehetséges scriptből, pl:
function formatTableStyle() {
    const table = document.getElementById('tableId');
    const cells = table.getElementsByClassName('customCell');
    for (let i = 0; i < cells.length; i++) {
        let cell = cells[i];
        let cellStyle = cell.style;
        let cellValue = cell.textContent;
        if(cellValue > 700000){
            cellStyle.color = 'red';
        }
        else if(cellValue > 400000){
            cellStyle.color = 'yellow';
        }
        else{
            cellStyle.color = 'green';
        }
    }
}
formatTableStyle();
